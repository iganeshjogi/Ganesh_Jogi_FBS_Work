Q1) Procedure with IN Parameters (Create New Account) 

Write a procedure named Create_Account that accepts: 
	- account_no (IN) 
	- customer_name (IN) 
	- opening_balance (IN) 
	- account_type (IN) 

Rules: 
	- Account number must be unique (should not already exist) 
	- Opening balance must be greater than or equal to 0 
	- Account type must be either 'Saving' or 'Current' 

Transaction Control: 

	- If all validations are satisfied, insert the record and COMMIT 
	- If validation fails, do not insert data, ROLLBACK and display proper message

Ans:

mysql> create procedure create_account(
    ->  in p_account_no int,
    ->  in p_customer_name varchar(50),
    ->  in p_opening_balance decimal (10,2),
    ->  in p_account_type varchar(30))
    -> begin
    ->
    -> declare cnt int default 0 ;
    ->
    -> select count(*)
    -> into cnt
    -> from bank_account
    -> where account_no = p_account_no;
    ->
    -> if cnt = 0
    -> and p_opening_balance >= 0
    -> and p_account_type in ('Saving','Current')
    -> then
    ->  insert into bank_account values
    ->  ( p_account_no, p_customer_name, p_opening_balance, p_account_type );
    -> commit ;
    ->
    -> else
    ->
    -> rollback ;
    ->
    -> select 'Invalid Account Details';
    ->
    -> end if;
    ->
    -> end #

	- If account not found, ROLLBACK and display message

mysql> create procedure withdraw_amount

    ->     in p_account_no int,
    ->     in p_withdraw_amount decimal(10,2)
    -> )
    ->
    -> begin
    ->
    -> declare cnt int default 0;

Q5) Procedure with Conditional TCL (Fund Transfer) 

Write a procedure named Transfer_Funds that accepts: 
	- from_account (IN) 
	- to_account (IN) 
	- transfer_amount (IN) 

Rules: 

	- Validate both accounts exist 
	- Transfer amount must be greater than 0 
	- Sender must have sufficient balance 
	- Deduct from sender and add to receiver 
	- COMMIT only if both updates succeed 
	- ROLLBACK if any validation fails or error occurs 

Ans:

mysql> create procedure transfer_funds
    -> (in from_account int, in to_account int, in transfer_amount decimal(10,2))
    ->
    -> begin
    ->
    -> declare cnt1 int default 0;
    -> declare cnt2 int default 0;
    -> declare from_balance decimal (10,2);
    ->
    -> select count(*) into cnt1
    -> from bank_account
    -> where account_no = from_account ;
    ->
    -> select count(*) into cnt2
    -> from bank_account
    -> where account_no = to_account ;
    ->
    -> if cnt1 = 1 and cnt2 = 1 then
    ->
    ->  select balance into from_balance
    ->  from bank_account
    ->  where account_no = from_account;
    ->
    ->  if transfer_amount <= from_balance then
    ->
    ->  update bank_account
    ->  set balance = balance - transfer_amount
    ->  where account_no = from_account;
    ->
    ->  update bank_account
    ->  set balance = balance + transfer_amount
    ->  where account_no = to_account;
    ->
    ->  commit;
    ->
    ->  select 'Successfully Transfered';
    ->
    ->  else
    ->
    ->  rollback;
    ->
    ->  select 'Insufficient balance';
    ->
    ->  end if;

    ->
    -> rollback;

Q8) Procedure with Conditional TCL (Close Account) 

Write a procedure named Close_Account that accepts: 
	- account_no (IN) 

Rules: 
	- Validate account exists 
	- Account can be closed only if balance is zero 
	- If balance is zero, delete record and COMMIT 
	- If balance is not zero, ROLLBACK and display message 
	- If account not found, ROLLBACK and display message

Ans:

mysql> delimiter #

mysql> create procedure close_account(in p_account_no int)
    -> begin
    ->
    -> declare cnt int default 0;
    -> declare bal decimal (10,2);
    ->
    ->  select count(*) into cnt
    ->  from bank_account
    ->  where account_no = p_account_no ;
    ->
    -> if cnt = 1 then
    ->
    ->  select balance into bal
    ->  from bank_account
    ->  where account_no = p_account_no;
    ->
    ->  if bal = 0 then
    ->          delete from bank_account
    ->          where account_no = p_account_no ;
    ->  commit;
    ->
    ->  select 'Account Deleted Successfully';
    ->
    ->  else
    ->          rollback;
    ->          select 'Balance Is Not Zero';
    ->  end if ;
    -> else
    ->  rollback;
    ->  select 'Account Not Found';
    ->
    -> end if;
    ->
    -> end #
